  • Bitwise operator
def dec2bin(number: int):
    ans = ""
    if ( number == 0 ):
        return 0
    while ( number ):
        ans += str(number&1)
        number = number >> 1

    ans = ans[::-1]

    return ans

def binToHexa(n):

    # convert binary to int
    num = int(n, 2)

    hex_num = hex(num)
